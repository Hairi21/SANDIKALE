// ======================= // DATA PRODUK // ======================= let daftarProduk = JSON.parse(localStorage.getItem("produk")) || []; function simpanProduk(){ localStorage.setItem("produk",JSON.stringify(daftarProduk)); } function tampilProduk(){ let tbody=document.getElementById("listProduk"); if(!tbody) return; tbody.innerHTML=""; daftarProduk.forEach((item,index)=>{ tbody.innerHTML+=` ${item.nama} ${item.kategori} Rp ${Number(item.harga).toLocaleString("id-ID")} `; }); } function tambahDataProduk(){ let nama=document.getElementById("namaProduk").value; let kategori=document.getElementById("kategoriProduk").value; let harga=document.getElementById("hargaProduk").value; if(nama=="" || kategori=="" || harga==""){ alert("Lengkapi data."); return; } daftarProduk.push({ nama:nama, kategori:kategori, harga:harga }); simpanProduk(); tampilProduk(); document.getElementById("namaProduk").value=""; document.getElementById("kategoriProduk").value=""; document.getElementById("hargaProduk").value=""; } function hapusProduk(index){ if(confirm("Hapus produk?")){ daftarProduk.splice(index,1); simpanProduk(); tampilProduk(); } } tampilProduk();
